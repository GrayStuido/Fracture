What is it?

FractureDisks or the .fracturedisk file is a way to embed and add your own games/websites to Fracture


Use

Burning a FractureDisk

To burn a .fracturedisk file you need to head on over to the Tools page and open up the Fracture Disk Burner. Once you are there you can burn a .fracturedisk file

Using The FractureDisk

Once you have a .fracturedisk file burned go to the Games page and look for Diskloader. Once you are there insert the disk and it'll load your game/page.

                document.addEventListener('DOMContentLoaded', function() {
                    const insertDiskButton = document.getElementById('insertDiskButton');
                    const fileInput = document.getElementById('fileInput');
                    const consoleElement = document.getElementById('console');
                    const cdContainer = document.getElementById('cdContainer');
                    const cdImage = document.getElementById('cdImage');
                    const upperContainer = document.querySelector('.upper-container');
        
                    insertDiskButton.addEventListener('click', function() {
                        fileInput.click();
                    });
        
                    fileInput.addEventListener('change', function(event) {
                        const file = event.target.files[0];
                        if (file && file.name.endsWith('.fracturedisk')) {
                            const reader = new FileReader();
                            reader.onload = function(e) {
                                const fileContent = e.target.result;
                                const separatorIndex = fileContent.indexOf('\n');
                                if (separatorIndex !== -1) {
                                    const diskInfoJson = fileContent.substring(0, separatorIndex).trim();
                                    const htmlContent = fileContent.substring(separatorIndex + 1).trim();
                                    
                                    try {
                                        const diskInfo = JSON.parse(diskInfoJson)[0];
                                        cdImage.src = diskInfo['ICON-PATH'];
                                        cdImage.alt = diskInfo.TITLE;
                                        
                                        simulateConsoleMessage('Loading disk: ' + diskInfo.TITLE);
                                        cdContainer.classList.add('spinning');
                                        
                                        setTimeout(() => {
                                            cdContainer.classList.remove('spinning');
                                            simulateConsoleMessage('Disk loaded successfully');
                                            simulateConsoleMessage('Inserting disk...');
                                            
                                            cdContainer.classList.add('inserting');
                                            
                                            setTimeout(() => {
                                                simulateConsoleMessage('Disk inserted');
                                                simulateConsoleMessage('Preparing to render HTML...');
                                                
                                                setTimeout(() => {
                                                    upperContainer.classList.add('shrink');
                                                    setTimeout(() => {
                                                        document.documentElement.innerHTML = htmlContent;
                                                    }, 1000);
                                                }, 1000);
                                            }, 1500);
                                        }, 3000);
                                    } catch (error) {
                                        console.error('Error parsing disk info JSON:', error);
                                        simulateConsoleMessage('Error: Invalid disk format', 'error');
                                    }
                                } else {
                                    simulateConsoleMessage('Error: Invalid .fracturedisk file format', 'error');
                                }
                            };
                            reader.readAsText(file);
                        } else {
                            simulateConsoleMessage('Error: Please upload a valid .fracturedisk file', 'error');
                        }
                    });
        
                    function simulateConsoleMessage(message, type = 'info') {
                        const messageElement = document.createElement('div');
                        messageElement.textContent = `> ${message}`;
                        messageElement.classList.add('console-message');
                        if (type === 'error') {
                            messageElement.style.color = '#ff4d6d';
                        }
                        consoleElement.appendChild(messageElement);
                        consoleElement.scrollTop = consoleElement.scrollHeight;
                    }
                });

Javascript Breakdown


Function Description Extra
document.addEventListener('DOMContentLoaded', function() {...}); Sets up the event listeners once the DOM content is fully loaded.
insertDiskButton.addEventListener('click', function() {...}); Triggers a click event on the file input when the insert disk button is clicked.
fileInput.addEventListener('change', function(event) {...}); Handles the change event when a file is selected, reads the file, and processes its content.
  • event: The event object containing information about the file input change.
reader.onload = function(e) {...}; Reads the content of the selected file and parses it as text.
  • e: The event object containing the result of the file read operation.
simulateConsoleMessage('Loading disk: ' + diskInfo.TITLE); Simulates a console message indicating that the disk is loading.
  • message: The message to display in the console.
  • type (optional): The type of message (info or error).
cdContainer.classList.add('spinning'); Adds a CSS class to the CD container to make it spin.
setTimeout(() => {...}, 3000); Sets a timeout to remove the spinning class after 3 seconds and proceed with further actions.
  • callback: The function to execute after the timeout.
  • delay: The delay in milliseconds.
cdContainer.classList.add('inserting'); Adds a CSS class to the CD container to simulate disk insertion.
upperContainer.classList.add('shrink'); Adds a CSS class to the upper container to shrink it, preparing for HTML rendering.
document.documentElement.innerHTML = htmlContent; Replaces the current HTML content of the document with the new HTML content from the file.
  • htmlContent: The HTML content to render.
console.error('Error parsing disk info JSON:', error); Logs an error message to the console if there is an issue parsing the disk info JSON.
  • error: The error object containing details about the JSON parsing issue.
simulateConsoleMessage('Error: Invalid disk format', 'error'); Simulates a console message indicating an error with the disk format.
  • message: The error message to display in the console.
  • type: The type of message (info or error).
simulateConsoleMessage('Error: Invalid .fracturedisk file format', 'error'); Simulates a console message indicating an invalid .fracturedisk file format.
  • message: The error message to display in the console.
  • type: The type of message (info or error).
simulateConsoleMessage('Error: Please upload a valid .fracturedisk file', 'error'); Simulates a console message indicating that the uploaded file is not a valid .fracturedisk file.
  • message: The error message to display in the console.
  • type: The type of message (info or error).
simulateConsoleMessage(message, type = 'info'); Appends a message to the console element on the page.
  • message: The message to display in the console.
  • type (optional): The type of message (info or error).
